void printPattern (int totalLines) {

    int printStar, updateValue, reverseUpdate, counter;

    for (int i = 1; i <= totalLines; i++) {

        printStar = (totalLines - i);
        updateValue = 1;
        reverseUpdate = ((2 * i) - 1);
        counter = i;

        for (int j = 1; j <= printStar; j++) {

            printf(" * ");
        }

        for (int k = 1; k <= ((2 * i) - 1); k++) {

            printf ("%2d ", counter);

            if (counter == reverseUpdate) {

                updateValue = - updateValue;
            }

            counter += updateValue;
        }

        for (int l = printStar; l >= 1; l--) {

            printf(" * ");
        }

        printf("\n");
    }
}

void printPattern (int totalLines) {

    int totalNumber, totalStar;

    for (int i = 1; i <= totalLines; i++) {

        totalStar = (totalLines - i);
        totalNumber = ((2 * i) - 1);

        printStar (totalStar);
        printNumberPattern (totalNumber);
        printStar (totalStar);

        cout << endl;
    }
}

void printStar (int total) {

    for (int i = 1; i <= total; i++) {

        cout << " * ";
    }
}

void printNumberPattern (int total) {

    int update = 1, changeUpdate = ((total + 1) / 2);

    for (int j = 1; j <= total; j++) {

        cout.width(2);
        cout << counter << " ";

        if (j == changeUpdate) {

            update *= -1;
        }

        counter += update;
    }

    counter += 2;
}

class pyramidpattern79 {

    static void printPattern (int totalLines) {

        int nextColumn = (totalLines - 1), printDigit;
        int itr = 1, counter;

        for (int i = totalLines; i >= 1; i--) {

            counter = itr;

            for (int j = 1; j <= totalLines; j++) {

                if (j >= i) {

                    System.out.printf("%2d ", counter++);
                } else {

                    System.out.print(" * ");
                }
            }

            counter -= 2;

            for (int k = 1; k <= nextColumn; k++) {

                if (k < itr) {

                    System.out.printf("%2d ", counter--);
                } else {

                    System.out.print(" * ");
                }
            }

            itr++;
            System.out.println();
        }

    }

        Scanner input = new Scanner (System.in);

        System.out.print("How many lines to be printed? --> ");

}
