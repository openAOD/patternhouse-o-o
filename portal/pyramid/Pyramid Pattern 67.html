void printPattern (int totalLines) {

    int whiteSpace, printStar;
    char alphabet = 'A';

    for (int i = 1; i <= totalLines; i++) {

        whiteSpace = (totalLines - i);
        printWhitespaces (whiteSpace);

        printStar = ((2 * i) - 1);

        for (int j = 1; j <= printStar; j++) {

            if (j & 1) {

                (i & 1) ? printf (" %c ", alphabet) : printf ("%2d ", i);
            } else {

                printf ("   ");
            }
        }

        alphabet++;
        printf("\n");
    }
}

import java.util.Scanner;

class pyramidpattern67 {

    static boolean ifNumber (int itr) {

        if ((itr & 1) != 0) {

            return false;
        } else {

            return true;
        }
    }

    static void printPattern (int totalLines) {

        int whiteSpaces, printStar, itr = 1;
        boolean number, character;

        char alphabet;

        for (int i = 1; i <= totalLines; i++) {

            number = ifNumber(itr);
            character = (!number);

            alphabet = (char)(64 + i);

            whiteSpaces = (totalLines - i);
            printStar = ((2 * i) - 1);

            for (int j = 1; j <= whiteSpaces; j++) {

                System.out.print("  ");
            }

            for (int k = 1; k <= printStar; k++) {

                if((k & 1) != 0 && number) {

                    System.out.printf("%3d", i);
                } else if ((k & 1) != 0 && character){

                    System.out.print("  " + alphabet);
                } else {

                    System.out.print(" ");
                }
            }

            itr++;
            System.out.println();
        }

    }

    public static void main(String[] args) {

        Scanner input = new Scanner (System.in);

        int totalLines;

        System.out.print("How many lines to be printed? --> ");
        totalLines = input.nextInt();

        System.out.println("\nYour pattern looks like this:\n");
        printPattern (totalLines);
    }
}
