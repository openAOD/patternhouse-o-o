void printPattern (int totalLines) {

    int columns, lastStar;
    columns = lastStar = ((2 * totalLines) - 1);

    for (int i = 1; i <= totalLines; i++) {

        for (int j = 1; j <= columns; j++) {

            if ((j == i) || (j == totalLines) || (j == (lastStar))) {

                cout << "* ";
            } else {

                cout << "0 ";
            }
        }

        lastStar--;
        cout << endl;
    }
}
