//Print "K"
#include <stdio.h>
#include <stdlib.h>
int main() {
    int height = 5;
    int i, j, half = height / 2, dummy = half;
    for (i = 0; i < height; i++) {
        printf("* ");
        for (j = 0; j <= half; j++) {
            if (j == abs(dummy)) {
                printf("* ");
            }
            else {
                printf("  ");
            }
        }
        printf("\n");
        dummy--;
    }
    return 0;
}

class symbolpattern {

    public static void main(String []args) {
        int n = 5, x = n/2 + 1;
        
        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= n/2 + 1; j++) {
                if(j == 1 || j == x) {
                    System.out.print("* ");
                }
                else {
                    System.out.print("  ");
                }
            }
            if(i <= n/2)
                x--;
            else
                x++;
                
            System.out.println();
        }
    }
}
