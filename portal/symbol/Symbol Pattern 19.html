#include <stdio.h>

int main()
{
    for (int i = 5; i >= 1; i--)
    {
        for (int j = 1; j <= (5 - i); j++)
        {
            printf("*");
        }
        for (int j = i; j >= 1; j--)
        {
            printf("%d", j);
        }

        printf("\n");
    }

    return 0;
}

void printPattern (int totalLines) {

    for (int i = totalLines; i >= 1; i--) {

        for (int j = totalLines; j >= 1; j--) {

            if (j <= i) {

                cout << j << " ";
            } else {

                cout << "* ";
            }
        }

        cout << endl;
    }
    
}

                if(j > i) {
                    System.out.print("* ");
