#include <stdio.h>
#include <stdlib.h>

int main()
{
    int n = 4, num = 1;
    for(int i = n; i >= -n; i--) {
        for(int j = 1; j <= abs(i); j++) 
            printf("  ");

        num = 1;
        
        for(int k = n; k >= abs(i); k--)
            printf("%d   ", num++);

        printf("\n");
    }
    return 0;
}

height = 4
num = 1

for i in range(height,-height-1,-1):
    
    for j in range(1,abs(i)+1):
        print(end="  ")
        
    for j in range(height,abs(i)-1,-1):
        print(height-j+1,end="   ")
            
    if(i > 0):
        num += 1
        
    else: 
        num -= 1
            
    print()
