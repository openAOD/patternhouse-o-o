#include <stdio.h>

int main()
{
    int n = 5, x = 1, y = n, num1, num2, t1, t2;
    for(int i = n; i >= 1; i--) {
        num1 = x;
        num2 = y;
        t1 = i;
        t2 = i + 1;
        for(int j = n; j >= i; j--) {
            if((i + j) % 2 == 0)
                printf("%2d ", num1);
            else
                printf("%2d ", num2);
            
            num1 = num1 - t1++;
            num2 = num2 - t2++;
        }
        y += i - 1;
        x += i;
        printf("\n");
    }
    return 0;
}

#include <iostream>
#include <vector>
using namespace std;

int main()
{
    vector<int> arr[5];
    int c = 1;
    bool flag = true;
    for (int i = 5; i >= 1; i--)
    {
        if (flag == true)
        {
            for (int j = 0; j < i; j++)
            {
                arr[j + 5 - i].push_back(c);
                c++;
            }
        }
        else
        {
            for (int j = 4; j >= (5 - i); j--)
            {
                arr[j].push_back(c);
                c++;
            }
        }
        flag = !flag;
    }
    for (int i = 0; i < 5; i++)
    {
        for (int j = arr[i].size() - 1; j >= 0; j--)
        {
            cout << arr[i][j] << " ";
        }
        cout << endl;
    }

    return 0;
}
