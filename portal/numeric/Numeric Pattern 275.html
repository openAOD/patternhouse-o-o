#include <stdio.h>

int main()
{
    int n = 3;
    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= i; j++) {
            for(int k = 1; k <= j; k++)
                printf("%d ", k);
                
            printf("\n");
        }
        printf("\n");
    }
    return 0;
}

    for(int i=0;i<n;i++){
        for(int j=0;j<=i;j++){
            cout<<j+1<<" ";
        }
        cout<<endl;
    }

}

public class Main
{
	
	public static void printTriangle(int n){
        for(int i=0;i<n;i++){
            for(int j=0;j<=i;j++)
                System.out.print(j+1 + " ");
            System.out.println();
        }
    }
	
	public static void main(String[] args) {
        int size=3;
    
        for(int i=1; i<size+1; i++){
            printTriangle(i);
            System.out.println();
        }
		
	}
