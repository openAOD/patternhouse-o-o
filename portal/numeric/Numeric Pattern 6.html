#include <stdio.h>

int main()
{
    int n,k=1;
    scanf("%d",&n);
    for(int i=n;i>=1;i--)
    {
        for(int j=n;j>=1;j--)
        {
            if(k>9)
            printf("%d ",k);
            else
            printf(" %d ",k);
            k=k+2;
        }
        printf("\n");
    }
    return 0;
}

i = 0
nums = list(range(1, 50, 2))
for _ in range(5):
    j = i + 5
    row = " ".join(map(str, nums[i:j]))
    print(row)
    i = j

class numericpattern6 {

    public static void main(String[] args) {
        int n = 5, num = 1;
        for (int i = 1; i <= n; i++){
            for (int j = 1; j <= n; j++) {
                System.out.printf("%2d ", num);
                num += 2;
            }
            System.out.println();
        }
    }
}
