int main()
{
    int n,k;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        k=i;
        for(int j=n;j>=1;j--)
        {
            if(k>9)
            printf("%d ",k);
            else
            printf(" %d ",k);
            k=k+i;
        }
        printf("\n");
    }
    return 0;
}

# Numeric Pattern 8

"""

1 2 3 4 5
2 4 6 8 10
3 6 9 12 15
4 8 12 16 20
5 10 15 20 25

"""

def pattern(rows):
    for x in range(1, rows + 1):
        row = " ".join(map(str, [x * i for i in range(1, rows + 1)]))
        print(row)

pattern(5)

class numericpattern8 {

    public static void main(String[] args) {
        int n = 5;
        for (int i = 1; i <= n; i++){
            for (int j = 1; j <= n; j++) {
                System.out.printf("%2d ", i*j);
            }
            System.out.println();
        }
    }
