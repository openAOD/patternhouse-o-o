#include <stdio.h>

int main()
{   int c=1;
    int m[5][5]={0};
    for (int i=0;i<5;i++)
    {
       for (int j=i;j<5;j++)
       {
        m[j][4-i]=c;
        c++;
       }
    }
    for (int i=0;i<5;i++)
    {
        for(int j=0;j<5;j++)
        {
            if (m[i][j])
            {
                printf("%3d",m[i][j]);
            }
            else
            {
                printf("   ");
            }
        }
        printf("\n");
    }

    return 0;
}

#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n = 5, x = 1, y, num;
    for(int i = n; i >= 1; i--) {
        y = i;
        num = x;

        for(int s = 1; s < i; s++) 
            cout << setw(3) << " ";
            
        for(int j = n; j >= i; j--){
            cout << setw(3) << num;
            num -= y;
            y++;
        }
        x+= i;
        cout << endl;
   }
   return 0;
}
