#include <stdio.h>

int main(){

    int n=3;

    int k=1;
    for(int i=0; i<n; i++){
        for(int j=0; j<=n-i-1; j++){
            printf("  ");
        }
        for(int j=n-i; j<=n; j++){
            printf("%d ",k);
        }
        printf("\n");
        k++;
    }
    for(int i=n; i>=0; i--){
        for(int j=0; j<=n-i-1; j++){
            printf("  ");
        }
        for(int j=n-i; j<=n; j++){
            printf("%d ",k);
        }
        printf("\n");
        k++;
    }
}

height = int(input())

        print(end="  ")

    for j in range(height-i,height):
        print(c,end=" ")

    c += 1
    print()

for i in range(1,height):

    for j in range(1,i+1):
        print(end="  ")

    for j in range(height-i,1,-1):
        print(c,end=" ")

    c += 1
    print()

        n/=2;
