#include <stdio.h>

int main()
{
    int c = 1;
    for (int i = 1; i <= 5; i++)
    {
        for (int j = 1; j <= (5 - i); j++)
        {
            printf(" ");
        }
        for (int j = c; j >= 1; j--)
        {
            printf("%d", j);
        }
        printf("\n");
        c = c + 2;
    }

    return 0;
}

def pattern(n):
    
    for i in range(n):
        count = 2*i+1
        for j in range(n - i - 1):
            print(' ', end='')
        for k in range(2 * i + 1):
            print(count, end='')
            count = count - 1
        print()

/**
 * Pattern100
 *     1
      321
     54321
    7654321
   987654321
 * 
 */
class numericpattern100 {
    public static void main(String[] args) {
        int rows = 5;
        
        for (int i = 1; i <= rows; i++) {
            for (int j = rows; j > i; j--) {
                System.out.print(" ");
            }
            for (int k = 2*i-1; k >= 1; k--) {
                System.out.print(k);
                
            }
            System.out.println();
        
        }
    }
}
