#include <stdio.h>

int main()
{
    int n,k=1,l;
    scanf("%d",&n);
    for(int i=n;i>=1;i--)
    {
        k=i;
        l=2*n-i+1;
        for(int j=1;j<=n;j++)
        {
            if(j%2!=0)
            {
            printf("%d ",k);
            k=k+2*n;
            }
            else
            {
            printf("%d ",l);
            l=l+2*n;
            }
        }
        printf("\n");
    }
    return 0;
}

class numericpattern14 {

    public static void main(String []args) {
        int n = 5, x, y;
        for(int i = 1; i <= n; i++) {
            x = i;
            y = n - i + 1;
            for(int j = 1; j <= n; j++) {
                if(j % 2 == 0) {
                    System.out.printf("%2d ", x);
                }
                else {
                    System.out.printf("%2d ", y);
                }
                
                x += n;
                y += n;
            }
            System.out.println();
        }
    }
    
}
