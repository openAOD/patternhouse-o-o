#include <iostream>

using namespace std;

int main()
{
    int N = (5 * 6) / 2;
    int fib[N];
    fib[0] = 0, fib[1] = 1;
    for (int i = 2; i <= N; i++)
    {
        fib[i] = fib[i - 1] + fib[i - 2];
    }
    for (int i = 1; i <= 5; i++)
    {
        int ind = 0;
        for (int j = 1; j <= i; j++)
        {
            cout << fib[ind++];
        }
        cout << endl;
    }

    return 0;
}

/**
 * pattern to be printed 
 * 
 *   0
 *   0 1
 *   0 1 1
 *   0 1 1 2
 *   0 1 1 2 3
 * numericpattern48
 * 
 */
class numericpattern48{
    public static void main(String[] args) {
        int rows = 5;
       numericPattern(rows);
    }
    private static void numericPattern(int n) {
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= i; j++) {
              System.out.print(fib(j-1)+"  ");
            }
            System.out.println();
        }
    } 
    private static int fib(int n) {
        if (n <= 1)
        return n;
     return fib(n-1) + fib(n-2);    
    } 
}
