#include <stdio.h>

int main()
{
    int n,k=1;
    scanf("%d",&n);
    for(int i=n;i>=1;i--)
    {
        k=i;
        for(int j=1;j<=n;j++)
        {
            printf("%d ",k);
            k=k+n;
        }
        printf("\n");
    }
    return 0;
}

class numericpattern13 {

    public static void main(String []args) {
        int n = 5, num;
        for(int i = 1; i <= n; i++) {
            num = n - i + 1;
            for(int j = 1; j <= n; j++) {
                System.out.printf("%2d ", num);
                num += n;
            }
            System.out.println();
        }
    }
    
}
