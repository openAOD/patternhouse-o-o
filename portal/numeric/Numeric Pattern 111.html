#include <stdio.h>
#include <math.h>

int main(){

    int n=3;

    int k=1;
    for(int i=0; i<n; i++){
        for(int j=0; j<n-i; j++){
            printf("  ");
        }
        int p=pow(2,i), sum=0;
        for(int j=0; j<p; j++){
            sum+=(k+(2*j));
        }
        for(int j=0; j<floor(p/2); j++){
            printf("%d ",k);
            k+=2;
        }
        if(sum>1)
            printf("%d ",sum);
        for(int j=p/2; j<p; j++){
            printf("%d ",k);
            k+=2;
        }

        printf("\n");
    }
}

class numericpattern111 {

    public static void main(String []args) {
        int n = 3, t = 1, x = 1, y;
        for(int i = 0; i < n; i++) {
            for(int j = n - 1; j > i; j--) {
                System.out.printf("  ");
            }
            for(int k = 0; k < t; k++) {
                if(i == k) {
                    y = 2*k*(x + 1);
                    y = (y == 0) ? 1 : y;
                    System.out.print(y + " ");
                    continue;
                }
                x += 2;
                System.out.print(x + " ");
            }
            
            t += 2;
            System.out.println();
        }
    }
    
}
