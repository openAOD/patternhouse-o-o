def pattern(n):
    alpha = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    countj = 0
    for i in range(0,n):
        counti = 0
        for j in range(0,n):
            flag = counti+countj
            if(flag>25):
                flag = flag - 26
            print(alpha[flag],end=" ")
            counti = counti+5
        print("\r")
        countj=countj+1
            
print("Enter the number of rows: ")
n = int(input())
pattern(n)

import java.util.Scanner;
public class AlphabeticPattern7
{
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int n = s.nextInt(); // Input no. of words to be printed in each line
        
        for(int i=0;i<n;i++) {
            for(int j=0;j<n;j++) {
                char c = (char)(i+j*5+65);
                System.out.print(c + " ");
            }
            System.out.println();
        }
    }
}

// Sample Input :- 5
// Output :-
// A F K P U
// B G L Q V
// C H M R W
// D I N S X
// E J O T Y
