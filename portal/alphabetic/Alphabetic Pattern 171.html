#include <stdio.h>

int main()
{
    int n = 5;
    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= n; j++) {
            if(j == 1 || i == 1 || i == n/2 + 1)
                printf("F ");
            else
                printf("  ");
        }
        printf("\n");
    }
    return 0;
}

#include <iostream>
using namespace std;
int main()
{
    int i, j, height = 5;
    for (i = 0; i < height; i++)
    {
        cout << "F";
        for (j = 0; j < height; j++)
        {
            if ((i == 0) || (i == height / 2 && j <= height / 2))
                cout << "F";
            else
                continue;
        }
        cout << "\n";
    }
    return 0;
}

class alphabeticpattern {

    public static void main(String []args) {
        int n = 5;
        
        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= n; j++) {
                if(i == 1 || i == n/2 + 1 || j == 1) {
                    System.out.print("F ");
                }
                else {
                    System.out.print("  ");
                }
            }
            System.out.println();
        }
    }
}
