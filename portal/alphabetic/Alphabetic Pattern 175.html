#include <stdio.h>

int main()
{
    int n = 5, mid = n/2 + 1;
    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= n; j++) {
            if(i == 1 || (i == n && j < mid) || j == mid)
                printf("J ");
            else
                printf("  ");
        }
        printf("\n");
    }
    return 0;
}

#include <iostream>
using namespace std;

int main()
{
    int height = 5;
    int i, j;
    for (i = 0; i < height; i++)
    {
        for (j = 0; j < height; j++)
        {
            if(i == 0) cout<<"J ";
            else if (i == height - 1 && (j > 0 && j < height - 1))
                cout << "J";
            else if ((j == height - 1 && i != height - 1) || (i > (height / 2) - 1 && j == 0 && i != height - 1))
                cout << "J";
            else
                cout << " ";
        }
        cout << "\n";
    }
    return 0;
}

class alphabeticpattern {

    public static void main(String []args) {
        int n = 5, mid = n/2 + 1;
        
        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= n; j++) {
                if(i == 1 || j == mid || (i == n && j < mid)) {
                    System.out.print("J ");
                }
                else {
                    System.out.print("  ");
                }
            }
            System.out.println();
        }
    }
}
