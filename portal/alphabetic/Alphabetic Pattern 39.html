#include <stdio.h>

int main()
{
    int c = 1;
    for (int i = 1; i <= 5; i++)
    {
        for (int j = 1; j <= (5 - i); j++)
        {
            printf(" ");
        }
        for (int j = c; j >= 1; j--)
        {
            printf("%c", 64 + j);
        }
        c = c + 2;
        printf("\n");
    }

    return 0;
}

alpha = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
def pattern(n):
    count = 0
    for i in range(n):
        for j in range(n - i - 1):
            print(' ', end='')
        flag = 0
        for k in range(2 * i + 1):
            print(alpha[count+flag], end='')
            flag=flag-1
        print()
        count = count + 2


print("Enter the number of rows: ")
n = int(input())
pattern(n)

import java.util.Scanner;
class AlphabeticPattern39
{
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int n = s.nextInt(); // Input no. of lines to be printed
      
         for(int i=0;i<n;i++) {
            for(int j=1;j<n-i;j++) {
              
                System.out.print("  ");
            }
            for(int j=2*i+1;j>0;j--) {
                char c = (char)(64+j);
                System.out.print(c + " ");
            }
            System.out.println();
        }
    }
}
