#include <stdio.h>

int main()
{
    int n = 5, mid = n/2 + 1;
    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= n; j++) {
            if(j == 1 || i == 1 || i == n || (i == mid && j >= mid) || (j == n && i > mid))
                printf("G ");
            else
                printf("  ");
        }
        printf("\n");
    }
    return 0;
}

#include <iostream>
using namespace std;

int main(){
    int i, j;
    int height = 5;
    int width = 2*height-1;
    width--;
    for (i = 0; i < height; i++) {
        for (j = 0; j < width; j++) {
            if ((i == 0 || i == height - 1)
                && (j == 0 || j == width - 1))
                cout <<" ";
            else if (j == 0)
                cout <<"G";
            else if (i == 0 && j <= height)
                cout <<"G";
            else if (i == height / 2
                    && j > height / 2)
                cout <<"G";
            else if (i > height / 2
                    && j == width - 1)
                cout <<"G";
            else if (i == height - 1
                    && j < width)
                cout <<"G";
            else
                cout <<" ";
        }
        cout <<"\n";
    }
    return 0;
}
