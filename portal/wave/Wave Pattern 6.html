#include <stdio.h>

int main()
{
    int n = 4;
    int height = n - 1, length = n, pc = n - 1;

    for (int i = 0; i <= height; i++) {
        for (int j = 0; j <= 2 * height * length; j++) {
            if ((j % (height * 2) == pc) || (j % (height * 2) == height + i)) {
                char ch = 'Z' - j;
                printf("%c ", ch);
            }
            else
               printf("  ");
        }
        pc -= 1;
        printf("\n");
    }
}

    for y in range(0, (wH * wL * 2) + 1):
        if (y % (wH * 2) == ctrl or
                y % (wH * 2) == wH + x):
            print(chr(cp), end="")
        else:
            print(" ", end="")  # single space
        cp -= 1  # 6
